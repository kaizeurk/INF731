6        public class Facture 
7        { 
8            /* Declaration des constantes */ 
9            public  const double TPS                 = 0.05; 
10           public  const double TVQ                 = 0.09975; 
11           public  const string NOM_RACINE          = "Facture-"; 
12           private const int    NONBRE_DE_PAS_LIGNE = 80; 
13           private const string AUTEURS             = "EMMANUEl FANGUE"; 
14           public  const string Erreur_fichier      = "erreur 404"; 
15           private const int colonne1 = 13; 
16           private const int colonne2 = 2; 
17           private const int colonne3 = 36; 
18           private const int colonne4 = 7; 
19           private const int colonne5 = 5; 
20           private const int colonne6 = 7; 
21           private const int ligne_total = 72; 
22            
23           /* Declaration des attributs */ 
24           private List<Article> articles; 
25           private string nom_facture; 
26           private double montant_tps; 
27           private double montant_tvq; 
28           private double sous_total; 
29    
30           public Facture()

32               articles = new List<Article>(); 
33           }

35           public List<Article> Articles 
36           { 
37               get => articles; 
38               set => articles = value; 
39           }

41           public string NomFacture 
42           { 
43               get => nom_facture; 
44               set

46                   if (FichierFacture.isFileExist(value)) 
47                   { 
48                       nom_facture = value; 
49                       genererFacture(); 
50                   } 
51               } 
52           } 
53    
54           public void ajouterArticle(Article in_article) 
55           { 
56               if ((in_article.GetType() == typeof(Article))) 
57               { 
58                   articles.Add(in_article);  
59               } 
60           }

62           public void ajouterArticles(List<Article> in_articles) 
63           { 
64             articles.AddRange(in_articles); 
65           } 
66    
67           private void genererFacture() 
68           { 
69               Articles = FichierFacture.lectureFacture(NomFacture, articles); 
70    
71           } 
72    
73           public void calculerTaxe(Article in_article) 
74           { 
75               if (in_article.estNonTaxable()==false) 
76               { 
77                   montant_tps += formatMonnai(in_article.Quantite * (TPS * in_article.PrixUniaite )); 
78                   montant_tvq += formatMonnai(in_article.Quantite * (TVQ * in_article.PrixUniaite)); 
79               } 
80    
81           } 
82    
83           private string entete() 
84           { 
85               string entete = new string('_',NONBRE_DE_PAS_LIGNE); 
86               entete += '\n'; 
87               entete += "   Facture produite pour le fichier "+ NOM_RACINE + nom_facture +'\n'; 
88               entete += "   par "+ AUTEURS +'\n'; 
89               entete += new string('_',NONBRE_DE_PAS_LIGNE) +'\n'; 
90    
91               return entete; 
92    
93           } 
94    
95           private string ligneFacture(Article in_article) 
96           { 
97               string ligne_facture = ""; 
98               ligne_facture += in_article.NumeroArticle.PadRight(colonne1, ' '); 
99               ligne_facture += (in_article.Quantite + "").PadRight(colonne2, ' '); 
100              ligne_facture += in_article.Description.PadRight(colonne3, ' '); 
101              ligne_facture += (formatMonnai(in_article.PrixUniaite) + " ").PadLeft(colonne4, ' '); 
102              ligne_facture += ((in_article.TaxeCategorie == Article.TAXABLE)?in_article.TaxeCategorie + " ":" ").PadRight(colonne5, ' '); 
103              ligne_facture += (formatMonnai(in_article.PrixUniaite*in_article.Quantite)+" $").PadLeft(colonne6, ' '); ; 
104              ligne_facture = ligne_facture.PadLeft(ligne_total, ' '); 
105   
106              ligne_facture += "\n"; 
107   
108              return ligne_facture; 
109          } 
110   
111          private string lignesFacture(List<Article> in_articles) 
112          { 
113              string chaineDeSortie = ""; 
114              foreach (Article article in in_articles) 
115              { 
116                  chaineDeSortie += ligneFacture(article); 
117              } 
118   
119              return chaineDeSortie; 
120          } 
121   
122          private string getBlockSousTotal() 
123          { 
124              double sous_total = sousTotal();  
125              return ("Sous-Total : " + (formatMonnai(sous_total) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n"+ 
126                     ("       TPS : " + (formatMonnai(montant_tps) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n" + 
127                     ("       TVQ : " + (formatMonnai(montant_tvq) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n" + 
128                     ("     Total : " + (formatMonnai(montant_tps + montant_tvq  + sous_total) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n"; 
129          } 
130           
131          public string ToString() 
132          { 
133              string chaineDeSortie = entete(); 
134              
135              chaineDeSortie += lignesFacture(Articles); 
136              chaineDeSortie += new string('-',NONBRE_DE_PAS_LIGNE) +'\n'; 
137   
138              chaineDeSortie += getBlockSousTotal(); 
139              return chaineDeSortie; 
140          } 
141   
142          public double sousTotal() 
143          { 
144              double sous_total = 0; 
145              montant_tps = 0; 
146              montant_tvq = 0; 
147              foreach (Article article in Articles) 
148              { 
149                  sous_total += article.PrixUniaite * article.Quantite; 
150                  calculerTaxe(article); 
151              } 
152   
153              return formatMonnai(sous_total); 
154   
155          } 
156   
157          private double formatMonnai(double montant) 
158          { 
159              return Math.Round(montant, 2); 
160          }
