126          public void calculerTaxe(Article in_article) 
127          { 
128              if (in_article.estNonTaxable()==false) 
129              { 
130                  montant_tps += formatMonnai(in_article.Quantite * (TPS * in_article.PrixUniaite ));

148          private string ligneFacture(Article in_article) 
149          { 
150              string ligne_facture = ""; 
151              ligne_facture += in_article.NumeroArticle.PadRight(colonne1, ' '); 
152              ligne_facture += (in_article.Quantite + "").PadRight(colonne2, ' '); 
153              ligne_facture += in_article.Description.PadRight(colonne3, ' '); 
154              ligne_facture += (formatMonnai(in_article.PrixUniaite) + " ").PadLeft(colonne4, ' '); 
155              ligne_facture += ((in_article.TaxeCategorie == Article.TAXABLE)?in_article.TaxeCategorie + " ":" ").PadRight(colonne5, ' '); 
156              ligne_facture += (formatMonnai(in_article.PrixUniaite*in_article.Quantite)+" $").PadLeft(colonne6, ' '); ; 
157              ligne_facture = ligne_facture.PadLeft(ligne_total, ' '); 
158   
159              ligne_facture += "\n"; 
160   
161              return ligne_facture; 
162          }

164          private string lignesFacture(List<Article> in_articles) 
165          {

167              foreach (Article article in in_articles) 
168              { 
169                  chaineDeSortie += ligneFacture(article); 
170              } 
171   
172              return chaineDeSortie; 
173          }

175          private string getBlockSousTotal() 
176          { 
177              double sous_total = sousTotal();  
178              return ("Sous-Total : " + (formatMonnai(sous_total) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n"+ 
179                     ("       TPS : " + (formatMonnai(montant_tps) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n" + 
180                     ("       TVQ : " + (formatMonnai(montant_tvq) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n" + 
181                     ("     Total : " + (formatMonnai(montant_tps + montant_tvq  + sous_total) + " $").PadLeft(11,' ')).PadLeft(ligne_total,' ') + "\n"; 
182          }

184          public override string ToString() 
185          { 
186              string chaineDeSortie = entete(); 
187              
188              chaineDeSortie += lignesFacture(Articles); 
189              chaineDeSortie += new string('-',NONBRE_DE_PAS_LIGNE) +'\n'; 
190   
191              chaineDeSortie += getBlockSousTotal(); 
192              return chaineDeSortie; 
193          } 
194   
195          public double sousTotal() 
196          { 
197              double sous_total = 0; 
198              montant_tps = 0; 
199              montant_tvq = 0; 
200              foreach (Article article in Articles) 
201              { 
202                  sous_total += article.PrixUniaite * article.Quantite; 
203                  calculerTaxe(article); 
204              } 
205   
206              return formatMonnai(sous_total); 
207   
208          } 
209   
210          private double formatMonnai(double montant) 
211          { 
212              return Math.Round(montant, 2); 
213          }

356          static public void ecrireFacture(string nom_facture_entrant, string enregistrement) 
357          { 
358              try 
359              { 
360                  flux_entrant = new StreamWriter(CHEMIN_RELATIF + nom_facture_entrant); 
361                  flux_entrant.WriteLine(enregistrement); 
362                  flux_entrant.Close(); 
363   
364              } 
365              catch (Exception e) 
366              { 
367                  Console.WriteLine(e.Message); 
368                  throw; 
369              }

380              flux_entrant.WriteLine(enregistrement);

383          /** 
384           * <param name="nom_facture_sortant">nom du fichier venant du client.</param> 
385           * <param name="list">Collection d'article venant du client</param> 
386           */ 
387          static public List<Article> lectureFacture(string nom_facture_sortant, List<Article> list) 
388          { 
389              if(isFileExist(nom_facture_sortant)) 
390              { 
391                  flux_sortant = new StreamReader(CHEMIN_RELATIF + nom_facture_sortant); 
392                  string ligne_article; 
393                  string[] element; 
394                  double prix_unit; 
395                  int  ligne= 0; 
396                  int  qte; 
397                  while (!flux_sortant.EndOfStream) 
398                  { 
399                      ligne_article = flux_sortant.ReadLine(); 
400                      element = ligne_article.Split(SEPARATEUR); 
401                      ligne++; 
402                      if (element.Length != 5) 
403                      { 
404                          throw new Exception(String.Format("Erreur dans le fichier {0} format incorrect a la linge {1}",nom_facture_sortant,ligne)); 
405                      } 
406                      else if (element[0] == String.Empty) 
407                      { 
408                          throw new Exception(String.Format("Erreur dans le fichier {0} la colonne 1 est vide a la linge {1}",nom_facture_sortant,ligne)); 
409                      } 
410                      else if ((Article.TAXABLE != element[1].ToUpper() && Article.NON_TAXABLE != element[1].ToUpper())) 
411                      { 
412                          throw new Exception(String.Format("Erreur dans le fichier {0} a la colonne 2 doit-etre {2} ou {3} a la linge {1}",nom_facture_sortant,ligne,Article.NON_TAXABLE,Article.TAXABLE)); 
413                      } 
414                      else if (element[3] == String.Empty) 
415                      { 
416                          throw new Exception(String.Format("Erreur dans le fichier {0} la colonne 4 est vide a la linge {1}",nom_facture_sortant,ligne)); 
417                      } 
418                       
419                      else if (int.TryParse(element[2], out qte)==false) 
420                      { 
421                          throw new Exception(String.Format("Erreur dans le fichier {0} la colonne 3 doit-etre un entier {1}",nom_facture_sortant,ligne)); 
422                           
423                      } 
424                      else if (double.TryParse(element[4],NumberStyles.Float, CultureInfo.GetCultureInfo("tr-TR"), out prix_unit) == false) 
425                      { 
426                          throw new Exception(String.Format("Erreur dans le fichier {0} la colonne 5 doit-etre de type 0,00 a la ligne {1}",nom_facture_sortant,ligne)); 
427                      } 
428                      else 
429                      { 
430                          prix_unit = Math.Round(prix_unit,2); 
431                          Article un_article = new Article(element[0], element[1], qte, element[3], 
432                              (float)prix_unit); 
433                          list.Add(un_article); 
434                           
435                      } 
436                      
437                  } 
438                   
439                  flux_sortant.Close(); 
440              } 
441              return list; 
442               
443          }
